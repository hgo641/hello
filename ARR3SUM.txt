.MODEL SMALL
.STACK 100H
.DATA 
A_ARRAY   DW   5 DUP (?)
B_ARRAY	DW	5	DUP	(?)
C_ARRAY	DW	5	DUP	(?)

SUM_ARRAY	MACRO	A_ARRAY,B_ARRAY,C_ARRAY,N
	LOCAL	@TOP
	PUSH SI
	PUSH CX
	XOR SI,SI
	REPT N
	XOR AX,AX
	ADD AX,A_ARRAY[SI]
	ADD AX,B_ARRAY[SI]
	MOV C_ARRAY[SI],AX
	ADD SI,2
	ENDM
	POP CX
	POP SI
	ENDM
.CODE
MAIN   PROC
   MOV   AX,@DATA
   MOV   DS,AX
   MOV   AH,2
   MOV   DL,'?'
   INT   21H

     
   MOV   BX,5
   XOR SI,SI
   MOV CX,5
@INPUTA:
  
  CALL   INDECM
  MOV A_Array[SI],AX
   ADD	SI,2
   LOOP @INPUTA

    MOV   AH,2
   MOV   DL,0DH
   INT   21H
   MOV   DL,0AH
   INT   21H

   MOV CX,5
   XOR SI,SI
   @INPUTB:
  
  CALL   INDECM
  MOV B_Array[SI],AX
   ADD	SI,2
   LOOP @INPUTB

   SUM_ARRAY	A_ARRAY,B_ARRAY,C_ARRAY,5

   MOV CX,5
   XOR SI,SI
   @OUTPUTC:
	MOV AX,C_ARRAY[SI]
	CALL OUTDECM
	ADD SI,2
	LOOP @OUTPUTC
  
   MOV   AH,4CH
   INT   21H
MAIN   ENDP

INCLUDE      INDECM.ASM
INCLUDE      OUTDECM.ASM
   END   MAIN
