.MODEL   SMALL
.STACK   100H
.DATA

.CODE
MAIN   PROC
   MOV   AX,@DATA
   MOV   AH,2
      MOV   DL,'?'
      INT   21H

      MOV   CX,3
@WHILE:
   CALL   INDECM
   PUSH   AX
   LOOP   @WHILE

   MOV   AH,2
   MOV   DL,0DH
   INT   21H
   MOV   DL,0AH
   INT   21H

@OUT:
POP   AX
MOV   SI,AX 
POP   AX
MOV   BX,AX
MOV   AX,60
MUL   BX;AX
MOV   BX,SI
ADD   BX,AX
MOV   SI,BX

POP   AX
MOV   BX,AX
MOV   AX,3600
MUL   BX;AX
MOV   BX,SI
ADD   BX,AX
MOV   AX,BX

CALL OUTDECM
MOV   AH,4CH
INT   21H
MAIN   ENDP

INCLUDE   INDECM.ASM
INCLUDE   OUTDECM.ASM
END MAIN
