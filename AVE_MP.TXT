.MODEL   SMALL
.STACK   100H
.DATA   
VarA   DW   ?
ArrayA   DW   5   DUP   (?)
ArrayB   DW   5   DUP   (?)

.CODE
MAIN   PROC
   MOV   AX,@DATA
   MOV   DS,AX
   MOV   AH,2
   MOV   DL,'?'
   INT   21H

   MOV   CX,5
   XOR   BX,BX
   XOR   SI,SI

@INPUTA:
   CALL   INDECM
   MOV   ArrayA[SI],AX
   ADD   BX,AX
   ADD   SI,2
   LOOP   @INPUTA

   MOV   AH,2
   MOV   DL,0DH
   INT   21H
   MOV   DL,0AH
   INT   21H

   MOV   AX,BX
   MOV   BL,5
   DIV   BL
   MOV   VarA,AX

   MOV   CX,5
   XOR   SI,SI
@OUTPUT:
   MOV   AX,ArrayA[SI]
   ADD   AX,VarA
   MOV   DX,ArrayA[SI]
   SUB   DX,VarA
   IMUL   DX
   CALL   OUTDECM
   MOV   AH,2
   MOV   DL,' '
   INT   21H
   ADD   SI,2
   LOOP   @OUTPUT
   MOV   AH,4CH
   INT   21H
MAIN   ENDP
INCLUDE   INDECM.ASM
INCLUDE   OUTDECM.ASM
   END   MAIN
