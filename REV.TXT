.MODEL SMALL
.STACK 100H
.DATA
.CODE
MAIN PROC
   MOV   AH,2
   MOV   DL,'?'
   INT   21H

   MOV   CX,5
@WHILE:
   CALL   INDECM
   PUSH   AX
   LOOP   @WHILE

   MOV   AH,2
   MOV   DL,0DH
   INT   21H
   MOV   DL,0AH
   INT   21H

   MOV   CX,5
@WHILE1:
   POP   AX
   CMP AX,0
   JL @NEXT
@ONE:
   CALL   OUTDECM
   MOV AH,2
   MOV DL,' '
   INT 21H
   LOOP    @WHILE1

   MOV   AH,4CH
   INT   21H
@NEXT:
   NEG AX
   JMP @ONE
MAIN   ENDP
INCLUDE INDECM.ASM
INCLUDE OUTDECM.ASM
   END MAIN
