.MODEL SMALL
.STACK 100H
.DATA 
ArrayB   DW   7 DUP (?)

SMALL_ARRAY MACRO ARRAY,N
   LOCAL @TOP,@NEXT
   PUSH BX
   PUSH CX
   LEA BX,ARRAY
   MOV AX,[BX]
   MOV CX,N
   DEC CX
@TOP:
   ADD BX,2
   CMP [BX],AX
   JGE @NEXT
   MOV AX,[BX]
@NEXT:
   LOOP @TOP
   POP CX
   POP BX
   ENDM

.CODE
MAIN PROC
   MOV AX,@DATA
   MOV DS,AX
   MOV AH,2
   MOV DL,'?'
   INT 21H
   MOV CX,7
   XOR SI,SI
@INPUTD:
   CALL INDECM
   MOV ArrayB[SI],AX
   ADD SI,2
   LOOP @INPUTD
   MOV AH,2
   MOV DL,0DH
   INT 21H
   MOV DL,0AH
   INT 21H

   SMALL_ARRAY   ArrayB,7
   CALL OUTDECM
   MOV AH,4CH
   INT 21H
MAIN ENDP
INCLUDE INDECM.ASM
INCLUDE OUTDECM.ASM
   END MAIN
