.MODEL   SMALL
.STACK   100H
.DATA
VarA   DW   ?
VarB   DW   ?
.CODE
MAIN   PROC
   MOV   AX,@DATA
   MOV   DS,AX
   MOV   AH,2
   MOV   DL,'?'
   INT   21H

   CALL   INDECM
   MOV   VarA,AX

@NEXT:

   MOV   AX,9
   IMUL   VarA
   MOV   BX,AX
   MOV   BL,5
   DIV   BL
   MOV   VarB,AX
   ADD   VarB,32


   MOV   AH,2
   MOV   DL,0DH
   INT   21H
   MOV   DL,0AH
   INT   21H
   MOV   AX,VarB
   CALL   OUTDECM
   MOV   AH,4CH
   INT   21H
MAIN   ENDP
INCLUDE   INDECM.ASM
INCLUDE   OUTDECM.ASM
   END   MAIN
